Data Preprocessing¶

這邊題目要求產生k=[1,4]的polynomial feature，因此我使用了for迴圈來循環產生這4組的polynomial feature並儲存起來

ploy_feature = []

for i in range(1, 5):
    poly = PolynomialFeatures(i)
    temp = poly.fit_transform(x)
    temp = np.mat(temp)
    ploy_feature.append(temp)

for i in range(1, 5):
    a = ploy_feature[i-1]
    coeff = (a.T * a).I * (a.T * y)
    coeff = np.array(coeff.T)[0]
    coeff = np.flip(coeff)
    str = "f(x) = %.2f "% (coeff[0])
    legend = "$%.2f"% (coeff[0])
    t = 0
    for i in coeff[1:]:
        t += 1
        if i > 0:
            str += "+"
            legend += "+"
        str += "%.2f×x^%d " % (i, t)
        legend += "%.2f×x^%d_i" % (i, t)
    legend += "$"
    print(str)

    draw_x = np.linspace(-1.5, 1.0, 1000)
    todraw = np.polyval(coeff, draw_x)
    plt.plot(draw_x, todraw, label=legend)
